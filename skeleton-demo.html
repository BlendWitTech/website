How to Use Skeletons in Your Code:

<!-- Card Skeleton Example -->
<div class="service-card">
    <div class="skeleton skeleton-image"></div>
    <div class="skeleton skeleton-title"></div>
    <div class="skeleton skeleton-text"></div>
    <div class="skeleton skeleton-text" style="width: 70%;"></div>
</div>

<!-- Replace with actual content when loaded -->
<div class="service-card" style="display: none;" id="actualContent">
    <!-- Real content here -->
</div>

<script>
// When content is loaded
setTimeout(() => {
    document.querySelector('.skeleton').parentElement.style.display = 'none';
    document.getElementById('actualContent').style.display = 'block';
}, 2000);
</script>
